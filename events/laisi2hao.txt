news_event = {
	id = laisi2hao.1
	title = laisi2hao.1.t
	desc = laisi2hao.1.d
	picture = GFX_xiaweiyihebao
	
	major = yes
	
	trigger = {
		has_global_flag = ouvrir_les_nouvelles
		629 = { is_controlled_by = USA }
		629 = { is_owned_by = USA }
		JAP = { has_war_with = USA }
		NOT = { has_global_flag = xiaweiyihebao }
		JAP = { has_government = fascism }
		USA = { has_government = democratic }
		has_global_flag = fall_of_chongqing
		has_global_flag = sov_peace_ger
	}
	
	immediate = {
		set_global_flag = xiaweiyihebao
		hidden_effect = {
			if = {
				limit = {
				NOT = {
				has_country_flag = nuke_a_flag
				}							
				}
				set_country_flag = nuke_a_flag
			}
			JAP = {
				launch_nuke = { 
				   province = 4180 #will nuke this province if specified
				   state = 629 #use either province or state. if state is used it will prefer enemies first while picking a province to nuke. otherwise it will pick one of the neutrals
				   controller = JAP #if state and controller is specified, the effect will pick a province that is controlled by this tag
				   use_nuke = yes #will consume nuke if specified
				}
			}
			play_song = japan_the_atomic_era
		}
	}
	
	mean_time_to_happen = {
		days = 7
	}
	
	option = {
		name = laisi2hao.1.a
		trigger = {
			NOT = {
				is_in_faction_with = USA
				is_in_faction_with = JAP
			}
		}
			hidden_effect = { play_song = japan_the_atomic_era }
		

	}
	option = {
		name = laisi2hao.1.b
		trigger = { is_in_faction_with = JAP }
		add_war_support = 0.1
		hidden_effect = { play_song = japan_the_atomic_era }
	}
	option = {
		name = laisi2hao.1.c
		trigger = { is_in_faction_with = USA }
		add_war_support = -0.1
		hidden_effect = { play_song = japan_the_atomic_era }
	}
}
